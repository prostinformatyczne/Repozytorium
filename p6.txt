void wczytywanie(int a[],int n)
{
        int i,k;
        for(i=0;i<n;i++)
                do
                {
                        printf("\n element %d = ",i);
                        k=scanf("%d",&a[i]);
                        fflush(stdin);
                }
                while(k==0);
}
void drukowanie(int a[],int n)
{
        int i;
        for(i=0;i<n;i++)
                printf("\n element %d = %d",i,a[i]);
}
